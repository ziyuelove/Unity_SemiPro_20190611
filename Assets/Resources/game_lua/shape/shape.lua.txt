--声明，这里声明了类名还有属性，并且给出了属性的初始值。
--保存生物战斗属性数据，
shape= {}

--这句是重定义元表的索引，就是说有了这句，这个才是一个类。
shape.__index = shape

--构造体，构造体的名字是随便起的，习惯性改为New()
function shape:New(id) 
    local self = {};    --初始化self，如果没有这句，那么类所建立的对象改变，其他对象都会改变
    setmetatable(self, shape);  --将self的元表设定为Class
    return self;    --返回自身
end

function shape:init()
	self.Hp = 100;
	self.Atk = 10;
	self.Def = 2;
end
