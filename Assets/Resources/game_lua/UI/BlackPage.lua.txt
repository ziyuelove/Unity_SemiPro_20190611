--声明，这里声明了类名还有属性，并且给出了属性的初始值。
BlackPage = {}

--这句是重定义元表的索引，就是说有了这句，这个才是一个类。
BlackPage.__index = BlackPage
BlackPage.gameobect = nil

--构造体，构造体的名字是随便起的，习惯性改为New()
function BlackPage:New() 
    local self = {};    --初始化self，如果没有这句，那么类所建立的对象改变，其他对象都会改变
    setmetatable(self, BlackPage);  --将self的元表设定为Class
    return self;    --返回自身
end

function BlackPage:LuaInit(LuaTable)
	self.LuaTable = LuaTable;
end

--开始时调用
function BlackPage:Start(gameobject) 
	game.OpenPage = true
	self.gameobect = gameobject;
	self.gameobect.transform.localScale = Vector3.New(1,1,1);
	self.gameobect.transform.localPosition = Vector3.New(0,0,0);

	self.gameobect:GetComponent("LuaComponent"):OnTimer("OnClose",2);
	return gameobject;
end

function BlackPage:OnClose(gameobject)
	game.OpenPage = false
	UnityEngine.GameObject.Destroy(self.gameobect);
	return gameobject;
end


function BlackPage.Update()
end




return BlackPage

--endregion
