

monsterAI = {}
--这句是重定义元表的索引，就是说有了这句，这个才是一个类。
monsterAI.__index = monsterAI
monsterAI.gameobect = nil


--构造体，构造体的名字是随便起的，习惯性改为New()
function monsterAI:New(gameObject) 
    local self = {};    --初始化self，如果没有这句，那么类所建立的对象改变，其他对象都会改变
    setmetatable(self, monsterAI);  --将self的元表设定为Class
	self.gameObject = gameObject;
	Debug.Log(gameObject.name.."  New")
    return self;    --返回自身
end

--开始时调用
function monsterAI:Start(gameobject) 
	self:Init();
end

function monsterAI:Init()
end
